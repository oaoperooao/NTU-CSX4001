Tweet collect and produce wordcloud

#Grab tweets
hmt <- search_tweets(
  "#undertale", n = 10000, include_rts = FALSE
)
## Searching for tweets...
## Finished collecting tweets!
#Unnest the words - code via Tidy Text
hmtTable <- hmt %>% 
  unnest_tokens(word, text)

#remove stop words - aka typically very common words such as "the", "of" etc
data(stop_words)
hmtTable <- hmtTable %>%
  anti_join(stop_words)
## Joining, by = "word"
#do a word count
hmtTable <- hmtTable %>%
  count(word, sort = TRUE) 
#Remove other nonsense words
hmtTable <-hmtTable %>%
  filter(!word %in% c('t.co','https','undertale',"art","it's",'el','en','tv','','','','','','','','','','','','','','','','','','','','','','','','','2','','','','<U+307E>','<U+307E><U+3059>','<U+307F>','3','<U+304B><U+3089>',"",'','','','','','','','','day','','',''))

wordcloud(words = hmtTable$word, freq = hmtTable$n, min.freq = 50,
          random.order=FALSE, rot.per=0.35,scale = c(3, 1),
          colors=brewer.pal(12,"Paired" ))
